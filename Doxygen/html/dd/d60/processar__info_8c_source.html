363 float saberBonus(Empresa *arrayRH, Lista_calc *conta, int posicao, int dia_max, int x) {
364  float tempo, idade, dias, bonus;
365  int dias_trab = conta->dados_calculo_array[x].dias_compl + conta->dados_calculo_array[x].dias_fds + conta->dados_calculo_array[x].dias_meios;
366 
367  tempo = 0.53 * (defineData(3) - arrayRH->funcionarios_array[posicao].entrada_emp.ano);
368  idade = 0.26 * ((defineData(3) - arrayRH->funcionarios_array[posicao].nascimento.ano) - 18);
369  if (dias_trab <= 22) {
370  dias = 0;
371  } else {
372  dias = (12.5 / (dia_max - 22)) * (dias_trab - 22);
373  }
374  bonus = (tempo + idade + dias) / 100;
375  return bonus;
376 }
